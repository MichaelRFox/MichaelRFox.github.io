// RGB conversions
import {invCompand} from '../utility/compand.js';
import {matrixMult} from '../utility/matrixMult.js';
import {w3cColors} from '../data/w3cColors.js';
import {trim} from '../utility/trim.js';

export function RGBtoHSV (rgbColor) {
    
    const r = rgbColor.r / 255;
    const g = rgbColor.g / 255;
    const b = rgbColor.b / 255;
    const min = Math.min(r, g, b);
    const max = Math.max(r, g, b);
    const δ = max - min;
    let h;
    
    if (δ === 0) return {h: 0, s: 0, v: 0}
    
    switch (max) {
        case r:
            h = 60 * (((g - b) / δ) % 6);
            break;
        case g:
            h = 60 * (2 + ((b - r) / δ));
            break;
        case b:
            h = 60 * (4 + ((r - g) / δ));
            break;
    };
    if (h < 0) h += 360;
    return {h: trim(h), s: trim(δ / max), v: trim(max)};
};

export function RGBtoHSL (rgbColor) {

    const r = rgbColor.r / 255;
    const g = rgbColor.g / 255;
    const b = rgbColor.b / 255;

    const min = Math.min(r, g, b);
    const max = Math.max(r, g, b);
    const δ = max - min;

    let h, s, l;
    
    l = trim(0.5 * (min + max));

    if (δ === 0) return {h: 0, s: 0, l: l};
    
    switch (max) {
        case r:
            h = 60 * (((g - b) / δ) % 6);
            break;
        case g:
            h = 60 * (2 + ((b - r) / δ));
            break;
        case b:
            h = 60 * (4 + ((r - g) / δ));
            break;
    };
    if (h < 0) h += 360;
    s = l < 0.5 ? δ / (max + min) : δ / (2 - max - min)
    // s = trim(δ / (1 - Math.abs((2 * l) -1)));
    // s = l <= 0.5 ? δ / 2 * l : δ / 2 - (2 * l);
    return {h: trim(h), s: s, l: l};
};

/**
 * @function RGBtoHWB
 * @desc converts an RGB object to an equivalent HWB object. Reference:
 * [W3C documentation]{@link https://www.w3.org/TR/css-color-4/#funcdef-hwb}
 * @param {Object} rgbColor An object on the form of {r: [0..255], g: [0..255], b: [0..255]}.
 * @returns {Object} an HWB object in the form of {h: [0..360], w: [0..1], b: [0..1]}.
 */
export function RGBtoHWB (rgbColor) {
  
    const r = rgbColor.r / 255;
    const g = rgbColor.g / 255;
    const b = rgbColor.b / 255; 

    const hsl = RGBtoHSL(rgbColor);
    const white = trim(Math.min(r, g, b));
    const black = trim(1 - Math.max(r, g, b));
    
    return {h: hsl.h, w: white, b: black};
}

export function RGBtoCMYK (rgbColor) {

    const r = rgbColor.r / 255;
    const g = rgbColor.g / 255;
    const b = rgbColor.b / 255; 
    
    const k = trim(1 - Math.max(r, g, b));
    const c = trim((1 - r - k) / (1 - k));
    const m = trim((1 - g - k) / (1 - k));
    const y = trim((1 - b - k) / (1 - k));
    
    return { c: c, m: m, y: y, k: k };
};

export function RGBtoXYZ (rgbColor, colorSpace) {
    const r = invCompand(rgbColor.r / 255.0, colorSpace.gammaRGB);
    const g = invCompand(rgbColor.g / 255.0, colorSpace.gammaRGB);
    const b = invCompand(rgbColor.b / 255.0, colorSpace.gammaRGB);
    const xyz = matrixMult ([r, g, b], colorSpace.rgb2xyz);
    return {x: trim(xyz[0]), y: trim(xyz[1]), z: trim(xyz[2])};
};

export function RBGtoLCHuv (rgbColor, colorSpace, observer) {
    return LUVtoLCHuv(XYZtoLUV(RGBtoXYZ(rgbColor, colorSpace), colorSpace, observer));
}

export function RGBtoHEX (rgbColor) {
    const r = String('00' + rgbColor.r.toString(16)).slice(-2);
    const g = String('00' + rgbColor.g.toString(16)).slice(-2);
    const b = String('00' + rgbColor.b.toString(16)).slice(-2);
    return (`#${r}${g}${b}`).toUpperCase();
};

export function HEXtoRGB (hexColor) {
    const r = parseInt(hexColor.slice(1, 3), 16);
    const g = parseInt(hexColor.slice(3, 5), 16);
    const b = parseInt(hexColor.slice(5, 7), 16);
    return {r: r, g: g, b: b};
}

export function RGBtoINT (rgbColor) {

    const r = rgbColor.r << 16;
    const g = rgbColor.g << 8;
    const b = rgbColor.b;

    return r + g + b;
}

export function INTtoRGB (int) {
    
    const r = int >> 16 & 0xFF;
    const g = int >> 8 & 0xFF;
    const b = int & 0xFF; 

    return {r: r, g: g, b: b};
}

export function RGBtoW3C (rgbColor) {

    const hexColor = RGBtoHEX(rgbColor);
    return Object.keys(w3cColors).find(key => w3cColors[key] === hexColor);
}

export function W3CtoRGB (w3cColor) {

    const hexColor = w3cColors[w3cColor.toLowerCase()];
    if (hexColor != undefined) return HEXtoRGB(hexColor);
}

/**
 * @function RGBtoLuminance
 * @desc Computes the luminance of an rgb color. See
 * {@link https://www.w3.org/TR/WCAG20-TECHS/G17.html#G17-procedure}
 * @param {Object} rgbColor An object representing and RGB color (e.g., {r: 255, g: 128, g: 0}).
 * @returns {Number} The relative luminance of the input color.
 */
export function RGBtoLuminance(rgbColor) {

    const a = [rgbColor.r, rgbColor.g, rgbColor.b].map(function (v) {
        v /= 255;
        return v <= 0.03928 ? v / 12.92 : Math.pow((v + 0.055) / 1.055, 2.4);
    });
    return a[0] * 0.2126 + a[1] * 0.7152 + a[2] * 0.0722;
}
