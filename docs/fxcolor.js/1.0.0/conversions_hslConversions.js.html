/**
 * @function HSLtoRGB
 * @desc Converts and HSL color to an RGB color. Source [W3C]{@link https://www.w3schools.com/lib/w3color.js}.
 * @param {Object} hslColor An HSL color object in the form of {h: [0..360], s: [0..1], l: [0..1]}.
 * @returns {Object} An RGB cobject in the form of {r: [0..255], g: [0..255], b: [0..255]}.
 */
export function HSLtoRGB (hslColor) {
  
  let temp1;

  const hue = hslColor.h / 60;
  
  if (hslColor.l <= 0.5) {
    temp1 = hslColor.l * (hslColor.s + 1);
  } else {
    temp1 = hslColor.l + hslColor.s - (hslColor.l * hslColor.s);
  }
  const temp2 = hslColor.l * 2 - temp1;
  
  const r = hueToRgb(temp2, temp1, hue + 2) * 255;
  const g = hueToRgb(temp2, temp1, hue) * 255;
  const b = hueToRgb(temp2, temp1, hue - 2) * 255;
  return {r : Math.round(r), g : Math.round(g), b : Math.round(b)};

}

function hueToRgb(temp2, temp1, hue) {

  if (hue < 0) hue += 6;
  if (hue >= 6) hue -= 6;
  if (hue < 1) return (temp1 - temp2) * hue + temp2;
  else if(hue < 3) return temp1;
  else if(hue < 4) return (temp1 - temp2) * (4 - hue) + temp2;
  else return temp2;
}
