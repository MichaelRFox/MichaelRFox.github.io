/**
 * @module validate
 * @desc Various functions to validate data input
 */

import {rgbColorSpaces} from '../data/rgbColorSpaces.js';

 export function validateColorSpace (colorSpace) {

    if (rgbColorSpaces.filter(d => d.name == colorSpace).length == 1) return true;
    console.error(`The input color Space, ${colorSpace} is not valid.`);
    return false;
 }

 export function validateRange (number, min = 0, max = 1) {

   if (typeof number != 'number' || number < min || number > max) {
      console.error(`The input, ${number} is invalid. It must be a number between ${min} an ${max}.`);
      return false;
   };
   return true;
 }
