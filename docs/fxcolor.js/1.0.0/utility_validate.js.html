/**
 * @module validate
 * @desc Various functions to validate data input
 */

import {rgbColorSpaces} from '../data/rgbColorSpaces.js';

/**
 * @function validateColorSpace
 * @desc Determines whether the input color space name is a valid RGB colorspace.
 * @param {string} colorSpace A string with the name of an RGB color space.
 * @returns {boolean} True if the color space exists, False otherwise.
 */
 export function validateColorSpace (colorSpace) {

    if (rgbColorSpaces.filter(d => d.name == colorSpace).length == 1) return true;
    console.error(`The input color Space, ${colorSpace} is not valid.`);
    return false;
 }

export function validateRange (number, min = 0, max = 1) {

   if (typeof number != 'number' || number < min || number > max) {
      console.error(`The input, ${number} is invalid. It must be a number between ${min} and ${max}.`);
      return false;
   };
   return true;
 }

export function validateArray (input, minLength = 3, maxLength = 4, ranges) {

   if (input instanceof Array && input.length >= minLength && input.length <= maxLength) {
      if (input.every((d, i) => validateRange(d, ranges[i].min, ranges[i].max))) {
         return true;
      };
   };
   return false;
 }


export function validateColorObject (input, keys, ranges) {
   if (typeof input == 'object') {
      const length = keys.length;
      let object = Object.entries(input).slice(0, length);
      object = object.reduce((a,b) => {a[b[0]] = b[1]; return a}, {});
      if (arrayEqual(Object.keys(object), keys)) {
         if (Object.values(object).every((d, i) => validateRange(d, ranges[i].min, ranges[i].max))) {
            return true;
         };
      };
   };
   return false;
 }

/**
 * @function arrayEqual
 * @desc Determines if two one-dimensional arrays are equal (order included).
 * @param {Array} a A one-dimensional array.
 * @param {Array} b A one-dimensional array.
 * @returns {boolean} True is the arrays are preciesely equal. False otherwise.
 */
function arrayEqual(a, b) {
   return a.every((d, i) => d == b[i]);
}

/**
 * @function validate
 * @desc Takes an input to be used to set the fxColor class object's{@link fxColor}
 * color, validates the input and returns the components and alpha value.
 * @param {(Array | Object | String)} input The input used to set the color. If it
 * is an array it should contain one element for each of the color components
 * (e.g., three for an RGB color or four for a CMYK color). If it is an object
 * it must have key/value pairs for each of the color components (e.g. {r: 255,
 * g: 128, b: 0}). If it is a string, it must be a properly formmated CSS color
 * specification (e.g., 'rgb(255, 128, 0)'. Optionally an additional value for
 * the alpha chabbel may be provides for each. Inthe case of an object the key must be 'alpha'.
 * @param {Array} keys An array of the keys for the color specification (e.g., ['r', 'g', 'b']).
 * The *alpha* key **must not** be incuded.
 * @param {Array} ranges An array of objects with key/value pairs containing the minimum and
 * maximum allowable values for each color component to include the alpha channel
 * (e.g., [{min:0, max:255}, {min:0, max:255}, {min:0, max:255}, {min:0, max:1}]).
 * @param {Function} parser The parse function to be used in the case of string input
 * (e.g., parseRgb).
 * @returns {Object} An object containing key/value pairs for the components
 * (an object with key/value pairs for the color components (e.g., {r: 255, g: 128, b: 0}))
 * and the alpha value. Either one of these elements may be *undefined* if either the input is
 * invalid or the alpha chanel value was not provided.
 */
 export function validate (input, keys, ranges, parser) {

   const length = keys.length;
   let components;
   let alpha;

   switch (Object.prototype.toString.call(input)) {
      case '[object Array]':
         const slicedInput = input.slice(0, length);
         if (slicedInput.every((d, i) => validateRange(d, ranges[i].min, ranges[i].max))) {
            components = slicedInput.reduce((a, b, i) => {a[keys[i]] = b; return a}, {});
            alpha = input[length];
         };
         break;
      case '[object Object]':
         const slicedKeys = Object.keys(input).slice(0, length);
         const slicedValues = Object.values(input).slice(0, length);
         if (arrayEqual(slicedKeys, keys) && slicedValues.every((d, i) => validateRange(d, ranges[i].min, ranges[i].max))) {
            components = slicedValues.reduce((a, b, i) => {a[keys[i]] = b; return a}, {});
            alpha = input.alpha;
         };
         break;
      case '[object String]':
         const parsedInput = parser(input);
         components = parsedInput != undefined ? Object.keys(parsedInput).reduce((a,b) => {if(b != 'alpha') {a[b] = parsedInput[b]}; return a},{}): parsedInput;
         alpha = parsedInput != undefined ? parsedInput.alpha : alpha;
         break;
   }

   if (alpha != undefined) {
      const range =  ranges[length - 1];
      alpha = validateRange(alpha, range.min, range.max) ? alpha : undefined;
   };

   return {components, alpha};
 }
